def compose(f, g):
    """–°–æ–∑–¥–∞–µ—Ç –∫–æ–º–ø–æ–∑–∏—Ü–∏—é —Ñ—É–Ω–∫—Ü–∏–π f(g(x))"""

# –ß–∞—Å—Ç–∏—á–Ω–æ–µ –ø—Ä–∏–º–µ–Ω–µ–Ω–∏–µ
def partial(func, *args):
    """–°–æ–∑–¥–∞–µ—Ç —Ñ—É–Ω–∫—Ü–∏—é —Å —á–∞—Å—Ç–∏—á–Ω–æ –ø—Ä–∏–º–µ–Ω–µ–Ω–Ω—ã–º–∏ –∞—Ä–≥—É–º–µ–Ω—Ç–∞–º–∏"""
    def wrapper(*remaining_args):
        return func(*(args + remaining_args))
    return wrapper

# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è

from typing import List, Dict, Optional, Callable, Union

def calculate_average(numbers: List[float]) -> float:
    """–í—ã—á–∏—Å–ª—è–µ—Ç —Å—Ä–µ–¥–Ω–µ–µ –∞—Ä–∏—Ñ–º–µ—Ç–∏—á–µ—Å–∫–æ–µ"""

def process_data(
    data: Dict[str, Union[int, str]], 
    processor: Callable[[str], str],
    default_value: Optional[str] = None
) -> Dict[str, str]:
    """–û–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç –¥–∞–Ω–Ω—ã–µ —Å –ø–æ–º–æ—â—å—é —Ñ—É–Ω–∫—Ü–∏–∏-–ø—Ä–æ—Ü–µ—Å—Å–æ—Ä–∞"""

    return result

    return principal * (1 + rate / compound_frequency) ** (compound_frequency * time)
