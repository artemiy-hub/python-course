```python

class LoggerMixin:
    """–ú–∏–∫—Å–∏–Ω –¥–ª—è –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏—è –¥–µ–π—Å—Ç–≤–∏–π"""
    
    def log_action(self, action):
        """–ó–∞–ø–∏—Å–∞—Ç—å –¥–µ–π—Å—Ç–≤–∏–µ –≤ –ª–æ–≥"""
        from datetime import datetime
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        print(f"[{timestamp}] {self.__class__.__name__}: {action}")

class SerializableMixin:
    """–ú–∏–∫—Å–∏–Ω –¥–ª—è —Å–µ—Ä–∏–∞–ª–∏–∑–∞—Ü–∏–∏ –æ–±—ä–µ–∫—Ç–æ–≤"""
    
    def to_dict(self):
        """–ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞—Ç—å –æ–±—ä–µ–∫—Ç –≤ —Å–ª–æ–≤–∞—Ä—å"""
        return {key: value for key, value in self.__dict__.items() 
                if not key.startswith('_')}
    
    def from_dict(self, data):
        """–í–æ—Å—Å—Ç–∞–Ω–æ–≤–∏—Ç—å –æ–±—ä–µ–∫—Ç –∏–∑ —Å–ª–æ–≤–∞—Ä—è"""
        for key, value in data.items():
            setattr(self, key, value)

class ValidationMixin:
    """–ú–∏–∫—Å–∏–Ω –¥–ª—è –≤–∞–ª–∏–¥–∞—Ü–∏–∏ –¥–∞–Ω–Ω—ã—Ö"""
    
    def validate(self):

        return True
    
    def is_valid(self):

class Electric:
    """–ö–ª–∞—Å—Å –¥–ª—è —ç–ª–µ–∫—Ç—Ä–∏—á–µ—Å–∫–∏—Ö —Ç—Ä–∞–Ω—Å–ø–æ—Ä—Ç–Ω—ã—Ö —Å—Ä–µ–¥—Å—Ç–≤"""
    
    def __init__(self, battery_capacity=100):
        self.battery_capacity = battery_capacity
        self.current_charge = battery_capacity
    
    def charge(self, amount=None):
        """–ó–∞—Ä—è–¥–∏—Ç—å –±–∞—Ç–∞—Ä–µ—é"""
        if amount is None:
            amount = self.battery_capacity - self.current_charge
        
        self.current_charge = min(self.battery_capacity, self.current_charge + amount)
        print(f"–ó–∞—Ä—è–¥ –±–∞—Ç–∞—Ä–µ–∏: {self.current_charge}/{self.battery_capacity}")
    
    def get_range(self):
        """–ü–æ–ª—É—á–∏—Ç—å –¥–∞–ª—å–Ω–æ—Å—Ç—å –ø–æ–µ–∑–¥–∫–∏"""
        return self.current_charge * 5  # 5 –∫–º –Ω–∞ –µ–¥–∏–Ω–∏—Ü—É –∑–∞—Ä—è–¥–∞

        Vehicle.__init__(self, make, model, year)

        if self.current_charge > 0:
            super().start()  # –í—ã–∑–æ–≤ –º–µ—Ç–æ–¥–∞ —Ä–æ–¥–∏—Ç–µ–ª—å—Å–∫–æ–≥–æ –∫–ª–∞—Å—Å–∞

            raise ValueError("–ï–º–∫–æ—Å—Ç—å –±–∞—Ç–∞—Ä–µ–∏ –¥–æ–ª–∂–Ω–∞ –±—ã—Ç—å –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω–æ–π")
        return True
```

```python
# –ê–±—Å—Ç—Ä–∞–∫—Ç–Ω—ã–π –±–∞–∑–æ–≤—ã–π –∫–ª–∞—Å—Å
class Shape(ABC):
    """–ê–±—Å—Ç—Ä–∞–∫—Ç–Ω—ã–π –∫–ª–∞—Å—Å –≥–µ–æ–º–µ—Ç—Ä–∏—á–µ—Å–∫–æ–π —Ñ–∏–≥—É—Ä—ã"""
    
    def __init__(self, color="white"):
        self.color = color
    
    @abstractmethod
    def area(self):
        """–ê–±—Å—Ç—Ä–∞–∫—Ç–Ω—ã–π –º–µ—Ç–æ–¥ –¥–ª—è –≤—ã—á–∏—Å–ª–µ–Ω–∏—è –ø–ª–æ—â–∞–¥–∏"""
        pass
    
    @abstractmethod
    def perimeter(self):
        """–ê–±—Å—Ç—Ä–∞–∫—Ç–Ω—ã–π –º–µ—Ç–æ–¥ –¥–ª—è –≤—ã—á–∏—Å–ª–µ–Ω–∏—è –ø–µ—Ä–∏–º–µ—Ç—Ä–∞"""
        pass

        """–û–±—â–∏–π –º–µ—Ç–æ–¥ –æ–ø–∏—Å–∞–Ω–∏—è —Ñ–∏–≥—É—Ä—ã"""
        return f"{self.color} {self.__class__.__name__}"
    
    # –ü–æ–ª–∏–º–æ—Ä—Ñ–Ω—ã–π –º–µ—Ç–æ–¥
    def scale(self, factor):
        """–ú–∞—Å—à—Ç–∞–±–∏—Ä–æ–≤–∞–Ω–∏–µ —Ñ–∏–≥—É—Ä—ã (–ø–µ—Ä–µ–æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç—Å—è –≤ –ø–æ–¥–∫–ª–∞—Å—Å–∞—Ö)"""
        print(f"–ú–∞—Å—à—Ç–∞–±–∏—Ä–æ–≤–∞–Ω–∏–µ {self.__class__.__name__} –≤ {factor} —Ä–∞–∑")

class Rectangle(Shape):
    """–ü—Ä—è–º–æ—É–≥–æ–ª—å–Ω–∏–∫"""
    
    def __init__(self, width, height, color="white"):
        super().__init__(color)
        self.width = width
        self.height = height
    
    def area(self):
        return self.width * self.height
    
    def perimeter(self):
        return 2 * (self.width + self.height)
    
    def scale(self, factor):
        super().scale(factor)
        self.width *= factor
        self.height *= factor
    
    def __str__(self):
        return f"Rectangle({self.width}x{self.height}, {self.color})"

class Circle(Shape):
    """–ö—Ä—É–≥"""
    
    def __init__(self, radius, color="white"):
        super().__init__(color)
        self.radius = radius
    
    def area(self):
        return math.pi * self.radius ** 2
    
    def perimeter(self):
        return 2 * math.pi * self.radius
    
    def scale(self, factor):
        super().scale(factor)
        self.radius *= factor
    
    def __str__(self):
        return f"Circle(radius={self.radius}, {self.color})"

class Triangle(Shape):
    """–¢—Ä–µ—É–≥–æ–ª—å–Ω–∏–∫"""
    
    def __init__(self, side_a, side_b, side_c, color="white"):
        super().__init__(color)
        self.side_a = side_a
        self.side_b = side_b
        self.side_c = side_c
    
    def area(self):
        # –§–æ—Ä–º—É–ª–∞ –ì–µ—Ä–æ–Ω–∞
        s = self.perimeter() / 2
        return math.sqrt(s * (s - self.side_a) * (s - self.side_b) * (s - self.side_c))
    
    def perimeter(self):
        return self.side_a + self.side_b + self.side_c
    
    def scale(self, factor):
        super().scale(factor)
        self.side_a *= factor
        self.side_b *= factor
        self.side_c *= factor
    
    def __str__(self):
        return f"Triangle({self.side_a}, {self.side_b}, {self.side_c}, {self.color})"

# –ü–æ–ª–∏–º–æ—Ä—Ñ–Ω–æ–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ
def print_shape_info(shape):
    """–§—É–Ω–∫—Ü–∏—è, —Ä–∞–±–æ—Ç–∞—é—â–∞—è —Å –ª—é–±–æ–π —Ñ–∏–≥—É—Ä–æ–π –ø–æ–ª–∏–º–æ—Ä—Ñ–Ω–æ"""
    print(f"–§–∏–≥—É—Ä–∞: {shape}")
    print(f"–û–ø–∏—Å–∞–Ω–∏–µ: {shape.describe()}")
    print(f"–ü–ª–æ—â–∞–¥—å: {shape.area():.2f}")
    print(f"–ü–µ—Ä–∏–º–µ—Ç—Ä: {shape.perimeter():.2f}")
    print("-" * 40)

shapes = [
    Rectangle(5, 3, "–∫—Ä–∞—Å–Ω—ã–π"),

    Triangle(3, 4, 5, "–∑–µ–ª–µ–Ω—ã–π")

    print(f"–ü–æ—Å–ª–µ –º–∞—Å—à—Ç–∞–±–∏—Ä–æ–≤–∞–Ω–∏—è: {shape}")
```

```python
    """–°–µ—Ä–≤–∏—Å —Å–µ—Ä–∏–∞–ª–∏–∑–∞—Ü–∏–∏"""
    
    @staticmethod
    def serialize_object(obj: Serializable) -> dict:
        """–°–µ—Ä–∏–∞–ª–∏–∑–æ–≤–∞—Ç—å –ª—é–±–æ–π –æ–±—ä–µ–∫—Ç, —Ä–µ–∞–ª–∏–∑—É—é—â–∏–π –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å Serializable"""
        if not isinstance(obj, Serializable):
            raise TypeError("–û–±—ä–µ–∫—Ç –¥–æ–ª–∂–µ–Ω —Ä–µ–∞–ª–∏–∑–æ–≤—ã–≤–∞—Ç—å –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å Serializable")
        
        return obj.serialize()
    
    @staticmethod
    def serialize_list(objects: list[Serializable]) -> list[dict]:
        """–°–µ—Ä–∏–∞–ª–∏–∑–æ–≤–∞—Ç—å —Å–ø–∏—Å–æ–∫ –æ–±—ä–µ–∫—Ç–æ–≤"""
        return [obj.serialize() for obj in objects if isinstance(obj, Serializable)]

class CacheService:
    """–°–µ—Ä–≤–∏—Å –∫–µ—à–∏—Ä–æ–≤–∞–Ω–∏—è"""
    
    def __init__(self):
        self._cache = {}
    
    def cache_object(self, obj: Cacheable) -> None:
        """–ó–∞–∫–µ—à–∏—Ä–æ–≤–∞—Ç—å –æ–±—ä–µ–∫—Ç"""
        if not isinstance(obj, Cacheable):
            raise TypeError("–û–±—ä–µ–∫—Ç –¥–æ–ª–∂–µ–Ω —Ä–µ–∞–ª–∏–∑–æ–≤—ã–≤–∞—Ç—å –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å Cacheable")
        
        key = obj.get_cache_key()
        ttl = obj.get_cache_ttl()
        
        # –ï—Å–ª–∏ –æ–±—ä–µ–∫—Ç —Ç–∞–∫–∂–µ —Å–µ—Ä–∏–∞–ª–∏–∑—É–µ–º—ã–π, —Å–æ—Ö—Ä–∞–Ω—è–µ–º —Å–µ—Ä–∏–∞–ª–∏–∑–æ–≤–∞–Ω–Ω—É—é –≤–µ—Ä—Å–∏—é
        if isinstance(obj, Serializable):
            self._cache[key] = {
                'data': obj.serialize(),
                'ttl': ttl,
                'cached_at': time.time()
            }
        else:
            self._cache[key] = {
                'data': obj,
                'ttl': ttl,
                'cached_at': time.time()
            }
        
        print(f"–û–±—ä–µ–∫—Ç –∑–∞–∫–µ—à–∏—Ä–æ–≤–∞–Ω —Å –∫–ª—é—á–æ–º: {key}, TTL: {ttl}s")
    
    def get_cached_object(self, cache_key: str) -> dict:
        """–ü–æ–ª—É—á–∏—Ç—å –æ–±—ä–µ–∫—Ç –∏–∑ –∫–µ—à–∞"""
        return self._cache.get(cache_key)

class ValidationService:
    """–°–µ—Ä–≤–∏—Å –≤–∞–ª–∏–¥–∞—Ü–∏–∏"""
    
    @staticmethod
    def validate_object(obj: Validatable) -> tuple[bool, list[str]]:
        """–í–∞–ª–∏–¥–∏—Ä–æ–≤–∞—Ç—å –æ–±—ä–µ–∫—Ç"""
        if not isinstance(obj, Validatable):
            raise TypeError("–û–±—ä–µ–∫—Ç –¥–æ–ª–∂–µ–Ω —Ä–µ–∞–ª–∏–∑–æ–≤—ã–≤–∞—Ç—å –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å Validatable")
        
        is_valid = obj.validate()
        errors = obj.get_validation_errors()
        
        return is_valid, errors
    
    @staticmethod
    def validate_batch(objects: list[Validatable]) -> dict:
        """–í–∞–ª–∏–¥–∏—Ä–æ–≤–∞—Ç—å —Å–ø–∏—Å–æ–∫ –æ–±—ä–µ–∫—Ç–æ–≤"""
        results = {}
        
        for i, obj in enumerate(objects):
            if isinstance(obj, Validatable):
                is_valid, errors = ValidationService.validate_object(obj)
                results[i] = {
                    'valid': is_valid,
                    'errors': errors,
                    'object_type': obj.__class__.__name__
                }
        
        return results

def demo_interfaces():
    """–î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è —Ä–∞–±–æ—Ç—ã —Å –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–∞–º–∏"""
    import time
    
    # –°–æ–∑–¥–∞–Ω–∏–µ –æ–±—ä–µ–∫—Ç–æ–≤
    user = User("u001", "alice", "alice@example.com", 25)
    product = Product("p001", "Laptop", 1000.0, "Electronics")
    invalid_user = User("u002", "ab", "invalid-email", -5)  # –ù–µ–≤–∞–ª–∏–¥–Ω—ã–π –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å
    
    objects = [user, product, invalid_user]
    
    print("=== –°–µ—Ä–∏–∞–ª–∏–∑–∞—Ü–∏—è ===")
    serialization_service = SerializationService()
    
    for obj in objects:
        try:
            serialized = serialization_service.serialize_object(obj)
            print(f"{obj.__class__.__name__}: {serialized}")
        except TypeError as e:
            print(f"–û—à–∏–±–∫–∞ —Å–µ—Ä–∏–∞–ª–∏–∑–∞—Ü–∏–∏: {e}")
    
    print("\n=== –ö–µ—à–∏—Ä–æ–≤–∞–Ω–∏–µ ===")
    cache_service = CacheService()
    
    for obj in objects:
        try:
            cache_service.cache_object(obj)
        except TypeError as e:
            print(f"–û—à–∏–±–∫–∞ –∫–µ—à–∏—Ä–æ–≤–∞–Ω–∏—è: {e}")
    
    print("\n=== –í–∞–ª–∏–¥–∞—Ü–∏—è ===")
    validation_service = ValidationService()
    
    validation_results = validation_service.validate_batch(objects)
    for idx, result in validation_results.items():
        print(f"–û–±—ä–µ–∫—Ç {idx} ({result['object_type']}):")
        print(f"  –í–∞–ª–∏–¥–µ–Ω: {result['valid']}")
        if result['errors']:
            print(f"  –û—à–∏–±–∫–∏: {result['errors']}")
    
    print("\n=== –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏—è –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–∞–º ===")
    for obj in objects:
        print(f"{obj.__class__.__name__}:")
        print(f"  Serializable: {isinstance(obj, Serializable)}")
        print(f"  Cacheable: {isinstance(obj, Cacheable)}")
        print(f"  Validatable: {isinstance(obj, Validatable)}")

demo_interfaces()
```
