    def _process_item(self, item: Dict) -> Dict:
        """–û–±—Ä–∞–±–æ—Ç–∞—Ç—å –æ–¥–∏–Ω —ç–ª–µ–º–µ–Ω—Ç (–ø—Ä–∏–≤–∞—Ç–Ω—ã–π –º–µ—Ç–æ–¥)"""

def process_data(data: List[Union[int, float]], 
                 operation: str = 'sum') -> Union[int, float]:
    """
    –û–±—Ä–∞–±–æ—Ç–∞—Ç—å —Å–ø–∏—Å–æ–∫ —á–∏—Å–ª–µ–Ω–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
    
    –§—É–Ω–∫—Ü–∏—è –≤—ã–ø–æ–ª–Ω—è–µ—Ç —Ä–∞–∑–ª–∏—á–Ω—ã–µ –æ–ø–µ—Ä–∞—Ü–∏–∏ –Ω–∞–¥ —Å–ø–∏—Å–∫–æ–º —á–∏—Å–µ–ª.
    
    Args:
        data: –°–ø–∏—Å–æ–∫ —á–∏—Å–µ–ª –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏
        operation: –¢–∏–ø –æ–ø–µ—Ä–∞—Ü–∏–∏ ('sum', 'mean', 'max', 'min')
        
    Returns:
        –†–µ–∑—É–ª—å—Ç–∞—Ç –æ–±—Ä–∞–±–æ—Ç–∫–∏ –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç –æ–ø–µ—Ä–∞—Ü–∏–∏
        
    Raises:
        DataError: –ï—Å–ª–∏ –¥–∞–Ω–Ω—ã–µ –Ω–µ–∫–æ—Ä—Ä–µ–∫—Ç–Ω—ã –∏–ª–∏ –æ–ø–µ—Ä–∞—Ü–∏—è –Ω–µ–∏–∑–≤–µ—Å—Ç–Ω–∞
        
    Example:
        >>> process_data([1, 2, 3, 4, 5], 'sum')
        15
        >>> process_data([1, 2, 3, 4, 5], 'mean')
        3.0
        
    Note:
        –§—É–Ω–∫—Ü–∏—è –ª–æ–≥–∏—Ä—É–µ—Ç –≤—Å–µ –æ–ø–µ—Ä–∞—Ü–∏–∏ –¥–ª—è –æ—Ç–ª–∞–¥–∫–∏
    """
    if not data:
        raise DataError("–°–ø–∏—Å–æ–∫ –¥–∞–Ω–Ω—ã—Ö –Ω–µ –º–æ–∂–µ—Ç –±—ã—Ç—å –ø—É—Å—Ç—ã–º", 1001)
    
    if not all(isinstance(x, (int, float)) for x in data):
        raise DataError("–í—Å–µ —ç–ª–µ–º–µ–Ω—Ç—ã –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å —á–∏—Å–ª–∞–º–∏", 1002)
    
    logger.info(f"–û–±—Ä–∞–±–æ—Ç–∫–∞ {len(data)} —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –æ–ø–µ—Ä–∞—Ü–∏–µ–π '{operation}'")
    
    operations = {
        'sum': sum,
        'mean': lambda x: sum(x) / len(x),
        'max': max,
        'min': min
    }
    
    if operation not in operations:
        available = ', '.join(operations.keys())
        raise DataError(
            f"–ù–µ–∏–∑–≤–µ—Å—Ç–Ω–∞—è –æ–ø–µ—Ä–∞—Ü–∏—è '{operation}'. "
            f"–î–æ—Å—Ç—É–ø–Ω—ã–µ: {available}", 
            1003
        )
    
    result = operations[operation](data)
    logger.info(f"–†–µ–∑—É–ª—å—Ç–∞—Ç –æ–ø–µ—Ä–∞—Ü–∏–∏ '{operation}': {result}")
    
    return result

class DataProcessor:
    """
    –ö–ª–∞—Å—Å –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ –¥–∞–Ω–Ω—ã—Ö —Å –Ω–∞—Å—Ç—Ä–∞–∏–≤–∞–µ–º—ã–º–∏ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º–∏
    
    Attributes:
        threshold: –ü–æ—Ä–æ–≥–æ–≤–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–ª—è —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏–∏
        normalize: –§–ª–∞–≥ –Ω–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏–∏ –¥–∞–Ω–Ω—ã—Ö
        
    Example:
        >>> processor = DataProcessor(threshold=0.5, normalize=True)
        >>> result = processor.process([1, 2, 3, 4, 5])
        >>> print(result)
        [0.0, 0.25, 0.5, 0.75, 1.0]
    """
    
    def __init__(self, threshold: float = 0.0, normalize: bool = False):
        """
        –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –ø—Ä–æ—Ü–µ—Å—Å–æ—Ä–∞
        
        Args:
            threshold: –ú–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–ª—è –≤–∫–ª—é—á–µ–Ω–∏—è –≤ —Ä–µ–∑—É–ª—å—Ç–∞—Ç
            normalize: –ù–æ—Ä–º–∞–ª–∏–∑–æ–≤–∞—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç –∫ –¥–∏–∞–ø–∞–∑–æ–Ω—É [0, 1]
        """
        self.threshold = threshold
        self.normalize = normalize
        self._processed_count = 0
        
        logger.info(f"–°–æ–∑–¥–∞–Ω DataProcessor —Å threshold={threshold}, "
                   f"normalize={normalize}")
    
    def process(self, data: List[Union[int, float]]) -> List[float]:
        """
        –û–±—Ä–∞–±–æ—Ç–∞—Ç—å –¥–∞–Ω–Ω—ã–µ —Å–æ–≥–ª–∞—Å–Ω–æ –Ω–∞—Å—Ç—Ä–æ–π–∫–∞–º
        
        Args:
            data: –°–ø–∏—Å–æ–∫ —á–∏—Å–µ–ª –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏
            
        Returns:
            –û–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫ —á–∏—Å–µ–ª
            
        Raises:
            DataError: –ü—Ä–∏ –Ω–µ–∫–æ—Ä—Ä–µ–∫—Ç–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
        """
        if not data:
            raise DataError("–°–ø–∏—Å–æ–∫ –¥–∞–Ω–Ω—ã—Ö –Ω–µ –º–æ–∂–µ—Ç –±—ã—Ç—å –ø—É—Å—Ç—ã–º")
        
        # –§–∏–ª—å—Ç—Ä–∞—Ü–∏—è –ø–æ –ø–æ—Ä–æ–≥—É
        filtered_data = [x for x in data if x >= self.threshold]
        
        if not filtered_data:
            logger.warning(f"–í—Å–µ –¥–∞–Ω–Ω—ã–µ –æ—Ç—Ñ–∏–ª—å—Ç—Ä–æ–≤–∞–Ω—ã –ø–æ—Ä–æ–≥–æ–º {self.threshold}")
            return []
        
        result = filtered_data.copy()
        
        # –ù–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è
        if self.normalize:
            min_val = min(result)
            max_val = max(result)
            
            if max_val > min_val:
                result = [(x - min_val) / (max_val - min_val) for x in result]
            else:
                result = [0.0] * len(result)
        
        self._processed_count += len(data)
        logger.info(f"–û–±—Ä–∞–±–æ—Ç–∞–Ω–æ {len(data)} —ç–ª–µ–º–µ–Ω—Ç–æ–≤, "
                   f"–ø–æ–ª—É—á–µ–Ω–æ {len(result)} —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤")
        
        return result
    
    def get_stats(self) -> dict:
        """
        –ü–æ–ª—É—á–∏—Ç—å —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫—É –æ–±—Ä–∞–±–æ—Ç–∫–∏
        
        Returns:
            –°–ª–æ–≤–∞—Ä—å —Å–æ —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–æ–π
        """
        return {
            'total_processed': self._processed_count,
            'threshold': self.threshold,
            'normalize': self.normalize
        }
    
    def reset_stats(self):
        """–°–±—Ä–æ—Å–∏—Ç—å —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫—É"""
        self._processed_count = 0
        logger.info("–°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ —Å–±—Ä–æ—à–µ–Ω–∞")

# –í—Å–ø–æ–º–æ–≥–∞—Ç–µ–ª—å–Ω—ã–µ —Ñ—É–Ω–∫—Ü–∏–∏ (–Ω–µ –≤ __all__)

    """–í–Ω—É—Ç—Ä–µ–Ω–Ω—è—è —Ñ—É–Ω–∫—Ü–∏—è –≤–∞–ª–∏–¥–∞—Ü–∏–∏ –¥–∞–Ω–Ω—ã—Ö"""
    return all(isinstance(x, (int, float)) for x in data)

def requires(*modules):
    """–î–µ–∫–æ—Ä–∞—Ç–æ—Ä –¥–ª—è –∏–º–ø–æ—Ä—Ç–∞ –º–æ–¥—É–ª–µ–π –ø—Ä–∏ –ø–µ—Ä–≤–æ–º –≤—ã–∑–æ–≤–µ"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            for module_name in modules:
                if module_name not in globals():
                    if module_name == 'pd':
                        import pandas as pd
                        globals()['pd'] = pd
                    elif module_name == 'np':
                        import numpy as np
                        globals()['np'] = np
                    # –î–æ–±–∞–≤—å—Ç–µ –¥—Ä—É–≥–∏–µ –º–æ–¥—É–ª–∏ –ø–æ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ—Å—Ç–∏
            
            return func(*args, **kwargs)
        return wrapper
    return decorator

@requires('pd', 'np')

    df = pd.DataFrame(data)
    array = np.array(data)
    return df.mean(), array.std()
