```python
import subprocess
import sys

def generate_requirements():
    """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è —Ñ–∞–π–ª–∞ requirements.txt"""
    result = subprocess.run([sys.executable, '-m', 'pip', 'freeze'], 
                          capture_output=True, text=True)
    
    with open('requirements.txt', 'w') as f:
        f.write(result.stdout)
    
    print("requirements.txt —Å–æ–∑–¥–∞–Ω")

def install_from_requirements():
    """–£—Å—Ç–∞–Ω–æ–≤–∫–∞ –ø–∞–∫–µ—Ç–æ–≤ –∏–∑ requirements.txt"""
    try:
        subprocess.run([sys.executable, '-m', 'pip', 'install', '-r', 'requirements.txt'], 
                      check=True)
        print("–ü–∞–∫–µ—Ç—ã —É—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω—ã —É—Å–ø–µ—à–Ω–æ")
    except subprocess.CalledProcessError as e:
        print(f"–û—à–∏–±–∫–∞ —É—Å—Ç–∞–Ω–æ–≤–∫–∏: {e}")

# –ü—Ä–æ–≥—Ä–∞–º–º–Ω–∞—è —É—Å—Ç–∞–Ω–æ–≤–∫–∞ –ø–∞–∫–µ—Ç–æ–≤
def install_package(package_name, version=None):
    """–ü—Ä–æ–≥—Ä–∞–º–º–Ω–∞—è —É—Å—Ç–∞–Ω–æ–≤–∫–∞ –ø–∞–∫–µ—Ç–∞"""
    if version:
        package = f"{package_name}=={version}"
    else:
        package = package_name
    
    try:
        subprocess.run([sys.executable, '-m', 'pip', 'install', package], 
                      check=True)
        print(f"–ü–∞–∫–µ—Ç {package} —É—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω")
        return True
    except subprocess.CalledProcessError:
        print(f"–û—à–∏–±–∫–∞ —É—Å—Ç–∞–Ω–æ–≤–∫–∏ {package}")
        return False

# –ü—Ä–æ–≤–µ—Ä–∫–∞ –¥–æ—Å—Ç—É–ø–Ω–æ—Å—Ç–∏ –ø–∞–∫–µ—Ç–∞
def check_package_available(package_name):
    """–ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞–ª–∏—á–∏—è –ø–∞–∫–µ—Ç–∞"""
    try:
        __import__(package_name)
        return True
    except ImportError:
        return False

# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è
if __name__ == "__main__":
    # –ü—Ä–æ–≤–µ—Ä—è–µ–º –Ω–∞–ª–∏—á–∏–µ requests
    if not check_package_available('requests'):
        print("–£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º requests...")
        install_package('requests')
    else:
        print("requests —É–∂–µ —É—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω")
```

### –ü—Ä–æ–≥—Ä–∞–º–º–Ω–æ–µ —É–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –≤–∏—Ä—Ç—É–∞–ª—å–Ω—ã–º–∏ –æ–∫—Ä—É–∂–µ–Ω–∏—è–º–∏

```python
import os
import sys
import subprocess
import venv
from pathlib import Path

class VirtualEnvironmentManager:
    """–ú–µ–Ω–µ–¥–∂–µ—Ä –≤–∏—Ä—Ç—É–∞–ª—å–Ω—ã—Ö –æ–∫—Ä—É–∂–µ–Ω–∏–π"""
    
    def __init__(self, envs_dir="venvs"):
        self.envs_dir = Path(envs_dir)
        self.envs_dir.mkdir(exist_ok=True)
    
    def create_env(self, env_name, python_version=None):
        """–°–æ–∑–¥–∞–Ω–∏–µ –≤–∏—Ä—Ç—É–∞–ª—å–Ω–æ–≥–æ –æ–∫—Ä—É–∂–µ–Ω–∏—è"""
        env_path = self.envs_dir / env_name
        
        if env_path.exists():
            print(f"–û–∫—Ä—É–∂–µ–Ω–∏–µ {env_name} —É–∂–µ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç")
            return False
        
        try:
            # –°–æ–∑–¥–∞–µ–º –æ–∫—Ä—É–∂–µ–Ω–∏–µ
            if python_version:
                subprocess.run([python_version, '-m', 'venv', str(env_path)], 
                             check=True)
            else:
                venv.create(env_path, with_pip=True)
            
            print(f"–û–∫—Ä—É–∂–µ–Ω–∏–µ {env_name} —Å–æ–∑–¥–∞–Ω–æ –≤ {env_path}")
            return True
        
        except Exception as e:
            print(f"–û—à–∏–±–∫–∞ —Å–æ–∑–¥–∞–Ω–∏—è –æ–∫—Ä—É–∂–µ–Ω–∏—è: {e}")
            return False
    
    def list_environments(self):
        """–°–ø–∏—Å–æ–∫ –≤–∏—Ä—Ç—É–∞–ª—å–Ω—ã—Ö –æ–∫—Ä—É–∂–µ–Ω–∏–π"""
        environments = []
        
        for env_dir in self.envs_dir.iterdir():
            if env_dir.is_dir():
                # –ü—Ä–æ–≤–µ—Ä—è–µ–º, —á—Ç–æ —ç—Ç–æ –≤–∏—Ä—Ç—É–∞–ª—å–Ω–æ–µ –æ–∫—Ä—É–∂–µ–Ω–∏–µ
                if (env_dir / 'pyvenv.cfg').exists():
                    environments.append(env_dir.name)
        
        return environments
    
    def get_env_python(self, env_name):
        """–ü–æ–ª—É—á–µ–Ω–∏–µ –ø—É—Ç–∏ –∫ Python –≤ –æ–∫—Ä—É–∂–µ–Ω–∏–∏"""
        env_path = self.envs_dir / env_name
        
        if os.name == 'nt':  # Windows
            python_path = env_path / 'Scripts' / 'python.exe'
        else:  # Unix/macOS
            python_path = env_path / 'bin' / 'python'
        
        return str(python_path) if python_path.exists() else None
    
    def install_package(self, env_name, package):
        """–£—Å—Ç–∞–Ω–æ–≤–∫–∞ –ø–∞–∫–µ—Ç–∞ –≤ –≤–∏—Ä—Ç—É–∞–ª—å–Ω–æ–µ –æ–∫—Ä—É–∂–µ–Ω–∏–µ"""
        python_path = self.get_env_python(env_name)
        
        if not python_path:
            print(f"–û–∫—Ä—É–∂–µ–Ω–∏–µ {env_name} –Ω–µ –Ω–∞–π–¥–µ–Ω–æ")
            return False
        
        try:
            subprocess.run([python_path, '-m', 'pip', 'install', package], 
                         check=True)
            print(f"–ü–∞–∫–µ—Ç {package} —É—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω –≤ {env_name}")
            return True
        except subprocess.CalledProcessError as e:
            print(f"–û—à–∏–±–∫–∞ —É—Å—Ç–∞–Ω–æ–≤–∫–∏: {e}")
            return False
    
    def run_in_env(self, env_name, command):
        """–í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ –∫–æ–º–∞–Ω–¥—ã –≤ –≤–∏—Ä—Ç—É–∞–ª—å–Ω–æ–º –æ–∫—Ä—É–∂–µ–Ω–∏–∏"""
        python_path = self.get_env_python(env_name)
        
        if not python_path:
            print(f"–û–∫—Ä—É–∂–µ–Ω–∏–µ {env_name} –Ω–µ –Ω–∞–π–¥–µ–Ω–æ")
            return False
        
        try:
            if isinstance(command, str):
                # –í—ã–ø–æ–ª–Ω—è–µ–º Python –∫–æ–¥
                result = subprocess.run([python_path, '-c', command], 
                                      capture_output=True, text=True)
            else:
                # –í—ã–ø–æ–ª–Ω—è–µ–º –∫–æ–º–∞–Ω–¥—É –∫–∞–∫ —Å–ø–∏—Å–æ–∫
                result = subprocess.run([python_path] + command, 
                                      capture_output=True, text=True)
            
            print(f"–í—ã–≤–æ–¥:\n{result.stdout}")
            if result.stderr:
                print(f"–û—à–∏–±–∫–∏:\n{result.stderr}")
            
            return result.returncode == 0
        
        except Exception as e:
            print(f"–û—à–∏–±–∫–∞ –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è: {e}")
            return False

# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è

    """–î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è –≤–∏—Ä—Ç—É–∞–ª—å–Ω—ã–º–∏ –æ–∫—Ä—É–∂–µ–Ω–∏—è–º–∏"""

    # –°–ø–∏—Å–æ–∫ –æ–∫—Ä—É–∂–µ–Ω–∏–π
    environments = manager.list_environments()
    print(f"\n–î–æ—Å—Ç—É–ø–Ω—ã–µ –æ–∫—Ä—É–∂–µ–Ω–∏—è: {environments}")
```

```python
def numpy_advanced():
    """–ü—Ä–æ–¥–≤–∏–Ω—É—Ç—ã–µ –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç–∏ NumPy"""

    a = np.array([[1, 2], [3, 4]])
    b = np.array([[5, 6], [7, 8]])
    
    print(f"\n–ú–∞—Å—Å–∏–≤ A:\n{a}")
    print(f"–ú–∞—Å—Å–∏–≤ B:\n{b}")

    # –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ –º–∞—Å—Å–∏–≤–æ–≤
    big_array = np.arange(16).reshape(4, 4)
    print(f"\n–ë–æ–ª—å—à–æ–π –º–∞—Å—Å–∏–≤:\n{big_array}")
    
    splits = np.split(big_array, 2, axis=0)
    print(f"–†–∞–∑–¥–µ–ª–µ–Ω–∏–µ –ø–æ —Å—Ç—Ä–æ–∫–∞–º: {len(splits)} —á–∞—Å—Ç–µ–π")
    for i, split in enumerate(splits):
        print(f"  –ß–∞—Å—Ç—å {i}:\n{split}")
    
    # –£—Å–ª–æ–≤–Ω—ã–µ –æ–ø–µ—Ä–∞—Ü–∏–∏
    data = np.random.randint(-10, 10, (3, 4))
    print(f"\n–ò—Å—Ö–æ–¥–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ:\n{data}")
    print(f"–ü–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è:\n{np.where(data > 0, data, 0)}")
    print(f"–ó–∞–º–µ–Ω–∞ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã—Ö –Ω–∞ -1:\n{np.where(data < 0, -1, data)}")
    
    # –õ–∏–Ω–µ–π–Ω–∞—è –∞–ª–≥–µ–±—Ä–∞
    matrix_a = np.random.rand(3, 3)
    matrix_b = np.random.rand(3, 3)

    print(f"–û–ø—Ä–µ–¥–µ–ª–∏—Ç–µ–ª—å A: {np.linalg.det(matrix_a):.4f}")
    
    try:

        print(f"–û–±—Ä–∞—Ç–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞ –Ω–∞–π–¥–µ–Ω–∞")

    except np.linalg.LinAlgError:

    print(f"–°–æ–±—Å—Ç–≤–µ–Ω–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è: {eigenvals}")
    
    # –†–µ—à–µ–Ω–∏–µ —Å–∏—Å—Ç–µ–º—ã –ª–∏–Ω–µ–π–Ω—ã—Ö —É—Ä–∞–≤–Ω–µ–Ω–∏–π Ax = b
```

```python
# –†–∞–±–æ—Ç–∞ —Å SciPy
try:
    import scipy
    from scipy import optimize, integrate, stats, signal
    from scipy.spatial.distance import cdist
    from scipy.interpolate import interp1d
    
    def scipy_examples():
        """–ü—Ä–∏–º–µ—Ä—ã —Ä–∞–±–æ—Ç—ã —Å SciPy"""
        
        # –û–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è
        print("=== –û–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è ===")
        
        # –ú–∏–Ω–∏–º–∏–∑–∞—Ü–∏—è —Ñ—É–Ω–∫—Ü–∏–∏
        def objective_function(x):
            return x**2 + 4*x + 4
        
        result = optimize.minimize_scalar(objective_function)
        print(f"–ú–∏–Ω–∏–º—É–º —Ñ—É–Ω–∫—Ü–∏–∏ x¬≤ + 4x + 4:")
        print(f"  x = {result.x:.4f}")
        print(f"  f(x) = {result.fun:.4f}")
        
        # –ü–æ–∏—Å–∫ –∫–æ—Ä–Ω–µ–π
        def equation(x):
            return x**3 - 2*x - 5
        
        root = optimize.fsolve(equation, 2)[0]
        print(f"–ö–æ—Ä–µ–Ω—å —É—Ä–∞–≤–Ω–µ–Ω–∏—è x¬≥ - 2x - 5 = 0: {root:.4f}")
        print(f"–ü—Ä–æ–≤–µ—Ä–∫–∞: f({root:.4f}) = {equation(root):.6f}")
        
        # –ò–Ω—Ç–µ–≥—Ä–∏—Ä–æ–≤–∞–Ω–∏–µ
        print("\n=== –ò–Ω—Ç–µ–≥—Ä–∏—Ä–æ–≤–∞–Ω–∏–µ ===")
        
        def integrand(x):
            return np.sin(x) * np.exp(-x)
        
        integral, error = integrate.quad(integrand, 0, np.inf)
        print(f"–ò–Ω—Ç–µ–≥—Ä–∞–ª sin(x)*exp(-x) –æ—Ç 0 –¥–æ ‚àû: {integral:.6f} ¬± {error:.2e}")
        
        # –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
        print("\n=== –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ ===")
        
        # –ì–µ–Ω–µ—Ä–∞—Ü–∏—è —Å–ª—É—á–∞–π–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
        data1 = stats.norm.rvs(loc=100, scale=15, size=100)
        data2 = stats.norm.rvs(loc=105, scale=12, size=100)
        
        # t-—Ç–µ—Å—Ç
        t_stat, p_value = stats.ttest_ind(data1, data2)
        print(f"t-—Ç–µ—Å—Ç –¥–ª—è –¥–≤—É—Ö –≤—ã–±–æ—Ä–æ–∫:")
        print(f"  t-—Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞: {t_stat:.4f}")
        print(f"  p-value: {p_value:.6f}")
        
        # –ö–æ—Ä—Ä–µ–ª—è—Ü–∏—è
        correlation, p_val = stats.pearsonr(data1[:50], data2[:50])
        print(f"–ö–æ—Ä—Ä–µ–ª—è—Ü–∏—è –ü–∏—Ä—Å–æ–Ω–∞: {correlation:.4f} (p={p_val:.4f})")
        
        # –û–±—Ä–∞–±–æ—Ç–∫–∞ —Å–∏–≥–Ω–∞–ª–æ–≤
        print("\n=== –û–±—Ä–∞–±–æ—Ç–∫–∞ —Å–∏–≥–Ω–∞–ª–æ–≤ ===")
        
        # –°–æ–∑–¥–∞–Ω–∏–µ —Å–∏–≥–Ω–∞–ª–∞ —Å —à—É–º–æ–º
        t = np.linspace(0, 1, 500)
        clean_signal = np.sin(2 * np.pi * 5 * t)  # 5 –ì—Ü
        noise = 0.3 * np.random.randn(len(t))
        noisy_signal = clean_signal + noise
        
        # –ü—Ä–∏–º–µ–Ω–µ–Ω–∏–µ —Ñ–∏–ª—å—Ç—Ä–∞
        b, a = signal.butter(4, 0.1, 'low')  # –ù–∏–∑–∫–æ—á–∞—Å—Ç–æ—Ç–Ω—ã–π —Ñ–∏–ª—å—Ç—Ä
        filtered_signal = signal.filtfilt(b, a, noisy_signal)
        
        # –û—Ü–µ–Ω–∫–∞ –∫–∞—á–µ—Å—Ç–≤–∞ —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏–∏
        mse_before = np.mean((noisy_signal - clean_signal)**2)
        mse_after = np.mean((filtered_signal - clean_signal)**2)
        
        print(f"MSE –¥–æ —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏–∏: {mse_before:.6f}")
        print(f"MSE –ø–æ—Å–ª–µ —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏–∏: {mse_after:.6f}")
        print(f"–£–ª—É—á—à–µ–Ω–∏–µ: {(mse_before/mse_after):.2f}x")
        
        # –ò–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü–∏—è
        print("\n=== –ò–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü–∏—è ===")
        
        x = np.array([0, 1, 2, 3, 4, 5])
        y = np.array([0, 1, 4, 9, 16, 25])  # y = x¬≤
        
        # –°–æ–∑–¥–∞–µ–º –∏–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü–∏–æ–Ω–Ω—É—é —Ñ—É–Ω–∫—Ü–∏—é
        f_linear = interp1d(x, y, kind='linear')
        f_cubic = interp1d(x, y, kind='cubic')
        
        # –¢–µ—Å—Ç–∏—Ä—É–µ–º –∏–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü–∏—é
        x_new = np.array([1.5, 2.5, 3.5])
        y_linear = f_linear(x_new)
        y_cubic = f_cubic(x_new)
        y_true = x_new**2
        
        print(f"–¢–æ—á–∫–∏ –¥–ª—è –∏–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü–∏–∏: {x_new}")
        print(f"–ò—Å—Ç–∏–Ω–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è: {y_true}")
        print(f"–õ–∏–Ω–µ–π–Ω–∞—è –∏–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü–∏—è: {y_linear}")
        print(f"–ö—É–±–∏—á–µ—Å–∫–∞—è –∏–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü–∏—è: {y_cubic}")
        
        # –†–∞—Å—Å—Ç–æ—è–Ω–∏—è –º–µ–∂–¥—É —Ç–æ—á–∫–∞–º–∏
        print("\n=== –†–∞—Å—Å—Ç–æ—è–Ω–∏—è ===")
        
        points = np.random.rand(5, 2) * 10  # 5 —Ç–æ—á–µ–∫ –≤ 2D
        distances = cdist(points, points, metric='euclidean')
        
        print(f"–ö–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã —Ç–æ—á–µ–∫:\n{points}")
        print(f"–ú–∞—Ç—Ä–∏—Ü–∞ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–π:\n{distances}")

    scipy_examples()

except ImportError:
```
